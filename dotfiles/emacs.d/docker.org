** Docker management
   #+begin_src emacs-lisp :tangle yes
     (use-package docker
       :defer t
       :config
       (setenv "PATH" (concat (getenv "PATH") ":/usr/local/bin"))
       (setq exec-path (append exec-path '("/usr/local/bin")))
       )
   #+end_src
